pydantic-settings>=2.0

# ──── Gemini multi-speaker TTS + image generation ────

# ──── SERP image search + image grading (poster-seed pipeline) ────

# ──── News ingestion (Phase 1d SP1) ────

trafilatura>=1.6

# ──── Audio Processing (needs ffmpeg/ffprobe on PATH) ────
pydub>=0.25

# ──── Acoustic caption alignment (offline Wav2Vec2 CTC forced alignment) ────
# torchaudio.functional.forced_align gives word-level karaoke timing from the
# known transcript + assembled audio — no paid API. CPU wheels are fine; the
# WAV2VEC2_ASR_BASE_960H model (~360MB) downloads once to the torch cache.
# num2words expands digits/currency to spoken words for the CTC label set.
# If these are absent the pipeline falls back to heuristic char-weight slicing.
torch>=2.4
torchaudio>=2.4
num2words>=0.5
numpy>=1.26

# ──── Persistence: Supabase service-role writes/uploads (Phase 1d SP3) ────
# supabase-py writes stories/digests/captions + uploads audio/poster to storage.
# python-dotenv loads .env for the standalone live e2e script (the agent clients
# read keys via pydantic-settings, which already bundles dotenv).
supabase>=2.0
python-dotenv>=1.0
# asyncpg: the catalog seeder (Phase 5f) writes via the IPv4 session pooler
# (scripts/seed_catalog/seed_via_pooler.py) because the supabase-py REST host
# is unreachable on IPv4-only networks (see memory news20-supabase-ddl-connection).
asyncpg>=0.29

# ship frequent releases; last release within weeks as of 2026-05).
